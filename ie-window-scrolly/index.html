function isIE() {
  const ua = window.navigator.userAgent;
  const msie = ua.indexOf("MSIE ");
  const trident = ua.indexOf("Trident/");
  return msie > 0 || trident > 0;
}
if (isIE()) {
  var fromTop = window.pageYOffset;
} else {
  var fromTop = window.scrollY;
}
